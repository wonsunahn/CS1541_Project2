77 case MemRead:
78 if (buffer.find(mreq->getAddr()) != buffer.end()) {
79 readHits.inc();
80 // The only latency added is the access delay of write buffer.
81 mreq->addLatency(hitDelay);
82 if(verbose) printf("%s->access(%s, addr: 0x%08x, latency: %u)\n", getName().c_str(), mreq->getMemOperationString().c_str(), mreq->getAddr(), mreq->getLatency());
83 } else {

86 // No delay added since write buffer access is concurrent with lower level access.
87 getLowerLevelMemObj()->access(mreq);

92 writeHits.inc();

96 // Send request to lower levels after marking as buffered write.
97 mreq->mutateToBufferedWrite();
98 getLowerLevelMemObj()->access(mreq);
99 // Update the expiration time of the pending store.
100 buffer[mreq->getAddr()].pendingUntil = cycle_number + mreq->getLatency() + mreq->getPendingCycles() - 1;
101 } else {

104 // The only latency added is the access delay of write buffer.
105 mreq->addLatency(hitDelay);

107 // Send request to lower levels after marking as buffered write.
108 mreq->mutateToBufferedWrite();
109 getLowerLevelMemObj()->access(mreq);
110 // Populate the write buffer with pending store.
111 buffer[mreq->getAddr()].pendingUntil = cycle_number + mreq->getLatency() + mreq->getPendingCycles() - 1;
112 } else {
113 writeOverflows.inc();
